### T2 · `src/services/chaining.ts` — new module (validation + helper builders)
**Path:** `src/services/chaining.ts` (new)
**Exports:**
- `type ExecutionTaskBlob` — exact shape per design §2.
- `type ChainResult` — `{kind:'posted', task_id, artifact_id} | {kind:'deferred', reason, deferred_event_id}`.
- `validateExecutionTask(blob: unknown): ExecutionTaskBlob | null`
  - Pure, throws ContextHubError('BAD_REQUEST', ...) on shape violation.
  - Constraints (per design §1+§2):
    - `title?`: string, ≤512 chars
    - `topology?`: one of TOPOLOGIES set
    - `slot?`: SLOT_REGEX (`^[a-z0-9][a-z0-9-]*$`), ≤64 chars
    - `kind?`: string, ≤64 chars, non-empty (no `(${something}).trim()===''`)
    - `depends_on?`: string[] of UUIDs (UUID_REGEX), max length 32 entries
    - `raci?`: object; `JSON.stringify(raci).length ≤ 8192`
- `buildChainedTaskParams(args): PostTaskParams` — pure, merges blob over derived defaults.
  - Defaults per design §2 "Chained task params resolution" table.
  - Slot default: `exec-${source_id.replace(/-/g,'').slice(0,16)}` (16 hex chars).
  - `raci` merge: `{...blobRaci, source_request|source_motion: source_id}` (system key wins).
**Verify:** `tsc --noEmit` clean; module compiles. No DB calls.

### T3 · `src/services/chaining.ts` — emitChain transactional helper
**Path:** same file, separate export
**Signature:**
```ts
export async function emitChain(
  client: PoolClient,
  args: {
    topic_id: string;
    source_event: { type: 'request.resolved' | 'motion.tallied'; source_id: string };
    actor_id: string;
    params: PostTaskParams;
  }
): Promise<ChainResult>;
```
**Code intent:**
1. `SELECT status FROM topics WHERE topic_id=$1 FOR UPDATE` — branch on result.
2. If `'closed'` or `'closing'`: build `would_be_task`, call `appendEvent` with type
   `'task.deferred'`, subject_type `'topic'`, subject_id `topic_id`, capture
   `event_id`, return `{kind:'deferred', reason, deferred_event_id}`.
3. If `'active'`:
   - Validate `depends_on` (chain-time): `SELECT count(*) FROM tasks WHERE task_id=ANY($1::uuid[]) AND topic_id=$2`. If count < depends_on.length, throw `ContextHubError('CHAINED_TASK_DEPENDENCY_INVALID', '...')`.
   - INSERT tasks row (`status='posted'`); INSERT artifacts row; INSERT artifact_versions row v1; appendEvent task.posted; appendEvent artifact.created.
   - Return `{kind:'posted', task_id, artifact_id}`.

// Sprint 15.7 — topology enforcement (DEFERRED-011b).
const taskMeta = await client.query<{ topology: string; depends_on: string[] }>(
  `SELECT topology, depends_on FROM tasks WHERE task_id = $1`,
  [taskId],
);
const { topology: taskTopology, depends_on: deps } = taskMeta.rows[0];
if (deps.length > 0) {
  if (taskTopology === 'sequential') {
    const depRes = await client.query<{ task_id: string; status: string }>(
      `SELECT task_id, status FROM tasks WHERE task_id = ANY($1::uuid[])`,
      [deps],
    );
    const incomplete = depRes.rows.filter((r) => r.status !== 'completed');
    if (incomplete.length > 0 || depRes.rowCount !== deps.length) {
      await client.query('ROLLBACK');
      return {
        status: 'unmet_dependencies',
        incomplete: incomplete.map((r) => ({task_id: r.task_id, status: r.status})),
        missing: deps.filter((d) => !depRes.rows.find((r) => r.task_id === d)),
      };

  } else if (taskTopology === 'rolling') {
    const upRes = await client.query<{ task_id: string; state: string }>(

      [deps],

    const not_baselined = upRes.rows.filter((r) => r.state !== 'baselined');
    if (not_baselined.length > 0) {
      await client.query('ROLLBACK');
      return {
        status: 'upstream_not_baselined',
        not_baselined: not_baselined.map((r) => ({task_id: r.task_id, state: r.state})),
      };

### T12 · `src/services/coordinationSweep.ts` — sweepStuckClosingTopics + scheduler wire-up
**Path:** `src/services/coordinationSweep.ts`
**Add:**
```ts
export const SWEEP_STALE_CLOSING_MINUTES = 5;
export const SWEEP_STUCK_CLOSING_STATEMENT_TIMEOUT_MS = 60_000;

export type StuckClosingSweepResult = { recovered: number; swept_at: string };

export async function sweepStuckClosingTopics(): Promise<StuckClosingSweepResult> {
  const pool = getDbPool();
  const stale = await pool.query<{ topic_id: string }>(
    `SELECT t.topic_id
       FROM topics t
       JOIN LATERAL (
         SELECT max(created_at) AS most_recent
           FROM coordination_events
          WHERE topic_id = t.topic_id AND type = 'topic.closing'
       ) ce ON true
      WHERE t.status = 'closing'
        AND ce.most_recent < now() - make_interval(mins => $1)`,
    [SWEEP_STALE_CLOSING_MINUTES],
  );
  let recovered = 0;
  for (const row of stale.rows) {
    try {
      await closeTopic({
        topic_id: row.topic_id,
        actor_id: 'system:closing-recovery',
        statementTimeoutMs: SWEEP_STUCK_CLOSING_STATEMENT_TIMEOUT_MS,
      });
      recovered++;
    } catch (err: any) {
      const msg = String(err?.code ?? err);
      if (msg.includes('57014')) {
        logger.warn({ topic_id: row.topic_id }, 'stuck_closing_timeout — retry next cycle');
      } else {
        logger.error({ topic_id: row.topic_id, err: String(err) }, 'stuck_closing_failed');
      }
    }
  }
  const result = { recovered, swept_at: new Date().toISOString() };
  logger.info(result, 'stuck-closing.sweep complete');
  return result;
}
```

logger.info({event:'stuck_closing_sweep_done', recovered: stuckSwept.recovered}, '...');
