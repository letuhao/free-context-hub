## Component design

### 1. `sweepExpiredLeases` (new service function)

`src/services/artifactLeases.ts` — append at the bottom:

```typescript
export type SweepResult = { rows_deleted: number; swept_at: string };

/**
 * Background sweep: delete leases that expired more than `grace_minutes` ago.
 * The grace window keeps recently-expired leases briefly as audit/debug
 * evidence (per design doc §"TTL enforcement"). Idempotent and
 * concurrency-safe — overlapping sweeps simply delete fewer rows.
 *
 * Default grace: 60 minutes. Lower bound: 0 (immediate). Upper bound: 1440 (24h).
 */
export async function sweepExpiredLeases(params?: { grace_minutes?: number }): Promise<SweepResult> {
  const grace = clampGrace(params?.grace_minutes);
  const pool = getDbPool();
  const r = await pool.query(
    `DELETE FROM artifact_leases
     WHERE expires_at < now() - make_interval(mins => $1)`,
    [grace],
  );
  const swept = { rows_deleted: r.rowCount ?? 0, swept_at: new Date().toISOString() };
  logger.info(swept, 'leases.sweep complete');
  return swept;
}

function clampGrace(graceMinutes?: number): number {
  if (graceMinutes === undefined || graceMinutes === null) return 60;
  if (graceMinutes < 0) return 0;
  if (graceMinutes > 1440) return 1440;
  return Math.floor(graceMinutes);
}
```

Notes:
- Uses `make_interval(mins => $1)` instead of `interval '$X minutes'` to avoid SQL injection via dynamic interval (parameter binding works correctly with this form).
- Returns rowCount + timestamp for caller logging.
- No transaction — single DELETE is atomic.
- Existing `sweep index` (`artifact_leases_expires_at_idx`) is used by the planner.
- **Unique-index reality (v2 correction):** migration 0048 ships a FULL UNIQUE index on `(project_id, artifact_type, artifact_id)` — partial-index design was abandoned because `now()` is not IMMUTABLE. Consequences for sweep semantics:
  - Expired-but-unswept rows occupy unique slots; `claim_artifact` on the same triple cleans them via step-1 DELETE inside its own transaction (already implemented in artifactLeases.ts:142-148).
  - The sweep is NOT a duplicate of the lazy cleanup — it is the **only** path that removes expired rows for artifacts no agent ever re-claims (long-tail abandoned leases). Without sweep, abandoned rows accumulate forever.
  - 60-min grace is preserved from master design (AC7 + design L234) — keeps freshly-expired rows briefly as a debugging convenience. Operationally negligible: at 10 sweeps/hour × ~hundreds of leases/day, table size stays bounded.
  - Concurrent step-1 DELETE (from a fresh claim) and sweep DELETE are safe: both target rows by primary key path, conflicts resolve to last-writer-wins with no observable effect.

### 3. Migration 0051 — extend `async_jobs.job_type` CHECK

`migrations/0051_leases_sweep_job_type.sql`:

```sql
-- Phase 13 Sprint 13.2 — add 'leases.sweep' to async_jobs.job_type CHECK constraint
-- Pattern: same as migration 0045 (document.extract.vision)
-- v2 (post-Adversary r1 WARN 3): defensive pre-migration assert prevents
-- silent clobber of unknown job_types added in intermediate migrations.

-- Pre-migration assertion (v3 — strengthened post-Adversary r2):
-- Verify the existing constraint contains EXACTLY the 13 types this migration
-- was designed against. Detects BOTH:
--   (a) missing types — design's expected set has something the DB doesn't (impossible in practice but defensive)
--   (b) EXTRA types — DB has a type that the design author didn't know about (clobber hazard)
-- Either case aborts loud with actionable error message.
DO $$
DECLARE
  current_def TEXT;
  expected_types TEXT[] := ARRAY[
    'repo.sync', 'workspace.scan', 'workspace.delta_index', 'index.run',
    'git.ingest', 'quality.eval', 'knowledge.refresh', 'faq.build',
    'raptor.build', 'knowledge.loop.shallow', 'knowledge.loop.deep',
    'knowledge.memory.build', 'document.extract.vision'
  ];
  parsed_types TEXT[];
  missing_type TEXT;
  extra_types TEXT[];
BEGIN
  SELECT pg_get_constraintdef(oid) INTO current_def
  FROM pg_constraint
  WHERE conname = 'async_jobs_job_type_check' AND conrelid = 'async_jobs'::regclass;

  IF current_def IS NULL THEN
    RAISE EXCEPTION 'async_jobs_job_type_check constraint not found — schema state unexpected';
  END IF;

  -- Parse the constraint definition. pg_get_constraintdef returns text like
  --   CHECK (job_type = ANY (ARRAY['x'::text, 'y'::text, ...]))
  -- Extract all single-quoted strings — each is a job_type literal.
  SELECT array_agg(m[1]) INTO parsed_types
  FROM regexp_matches(current_def, '''([^'']+)''', 'g') AS m;

  IF parsed_types IS NULL OR array_length(parsed_types, 1) IS NULL THEN
    RAISE EXCEPTION 'async_jobs_job_type_check could not be parsed: %', current_def;
  END IF;

  -- Check (a): every expected type must be present
  FOREACH missing_type IN ARRAY expected_types LOOP
    IF NOT (missing_type = ANY (parsed_types)) THEN
      RAISE EXCEPTION 'async_jobs_job_type_check missing expected type %; migration 0051 cannot safely proceed because the constraint shape has drifted. Inspect with: \d+ async_jobs', missing_type;
    END IF;
  END LOOP;

  -- Check (b): no extra types beyond what the design author knew about
  SELECT array_agg(t) INTO extra_types
  FROM unnest(parsed_types) AS t
  WHERE t <> ALL (expected_types);

  IF extra_types IS NOT NULL AND array_length(extra_types, 1) > 0 THEN
    RAISE EXCEPTION 'async_jobs_job_type_check has UNKNOWN type(s) the design did not include: %. Migration 0051 would silently drop them. Decide policy (preserve / migrate / drop) and update this migration explicitly.', extra_types;
  END IF;
END $$;

ALTER TABLE async_jobs DROP CONSTRAINT IF EXISTS async_jobs_job_type_check;
ALTER TABLE async_jobs ADD CONSTRAINT async_jobs_job_type_check
  CHECK (job_type IN (
    'repo.sync',
    'workspace.scan',
    'workspace.delta_index',
    'index.run',
    'git.ingest',
    'quality.eval',
    'knowledge.refresh',
    'faq.build',
    'raptor.build',
    'knowledge.loop.shallow',
    'knowledge.loop.deep',
    'knowledge.memory.build',
    'document.extract.vision',
    'leases.sweep'
  ));
```

### 5a. NEW (v4): `src/services/sweepScheduler.ts` — extracted from src/index.ts for testability

```typescript
import crypto from 'node:crypto';
import { getDbPool } from '../db/client.js';
import { enqueueJob } from './jobQueue.js';
import { createModuleLogger } from '../utils/logger.js';

const logger = createModuleLogger('sweep-scheduler');

export const SWEEP_INTERVAL_MS = 15 * 60 * 1000;  // 15 minutes

// v4 (post-Adversary r3 BLOCK 2): advisory-lock key derived deterministically
// from a stable string literal. First 8 bytes of SHA256 interpreted as signed
// int64 (Postgres advisory locks accept signed bigint).
// Registry: docs/operations/advisory-locks.md
export const LEASES_SWEEP_ADVISORY_KEY: bigint = (() => {
  const digest = crypto.createHash('sha256').update('phase-13.leases-sweep').digest();
  // First 8 bytes → signed int64 (read as bigint then coerce sign)
  const u64 = digest.readBigUInt64BE(0);
  // Convert to signed int64 range
  const SIGN_BIT = 1n << 63n;
  return u64 >= SIGN_BIT ? u64 - (1n << 64n) : u64;
})();

/**
 * Start the periodic sweep scheduler.
 * Returns an opaque handle for stopping the scheduler (test only — production
 * lets the SIGINT/exit clear the timer).
 *
 * Multi-replica safety: each cycle attempts to acquire a Postgres advisory
 * lock; only the lock-holder enqueues a leases.sweep job. Lock is released
 * after enqueue (not held during the actual sweep — that runs in the worker).
 */
export interface SweepHandle {
  stop(): void;
}

export function startSweepScheduler(opts?: {
  intervalMs?: number;          // override for tests
  graceMinutes?: number;        // payload override (default 60)
}): SweepHandle {
  const interval = opts?.intervalMs ?? SWEEP_INTERVAL_MS;
  const grace = opts?.graceMinutes ?? 60;
  let stopped = false;
  let timer: NodeJS.Timeout | null = null;

  const cycle = async () => {
    if (stopped) return;
    let gotLock = false;
    const pool = getDbPool();
    const client = await pool.connect().catch(() => null);
    try {
      if (!client) {
        logger.error({ event: 'leases_sweep_pool_unavailable' }, 'db pool unavailable; will retry');
        return;
      }
      const r = await client.query<{ acquired: boolean }>(
        `SELECT pg_try_advisory_lock($1::bigint) AS acquired`,
        [LEASES_SWEEP_ADVISORY_KEY.toString()],
      );
      gotLock = r.rows[0]?.acquired === true;
      if (!gotLock) {
        logger.info({ event: 'leases_sweep_skip_not_leader' }, 'another replica scheduled sweep this cycle');
        return;
      }
      const enq = await enqueueJob({
        job_type: 'leases.sweep',
        payload: { grace_minutes: grace },
        correlation_id: `sweep-${Date.now()}`,
      });
      logger.info({ event: 'leases_sweep_enqueued', job_id: enq.job_id, backend: enq.backend }, 'leases.sweep scheduled');
    } catch (err) {
      logger.error({ event: 'leases_sweep_enqueue_failed', err: err instanceof Error ? err.message : String(err) }, 'leases.sweep enqueue failed; will retry next interval');
    } finally {
      if (gotLock && client) {
        await client.query(`SELECT pg_advisory_unlock($1::bigint)`, [LEASES_SWEEP_ADVISORY_KEY.toString()]).catch(() => {});
      }
      if (client) client.release();
      // schedule next cycle (chained, not interval — avoids overlap)
      if (!stopped) timer = setTimeout(() => { cycle(); }, interval);
    }
  };

  timer = setTimeout(cycle, interval);
  logger.info({ interval_minutes: interval / 60_000, advisory_key: LEASES_SWEEP_ADVISORY_KEY.toString() }, 'leases.sweep scheduler started');

  return {
    stop() {
      stopped = true;
      if (timer) clearTimeout(timer);
    },
  };
}
```

```typescript
import { enqueueJob } from './services/jobQueue.js';
import { getDbPool } from './db/client.js';

const SWEEP_INTERVAL_MS = 15 * 60 * 1000;  // 15 minutes

// v3 (post-Adversary r2 BLOCK 2): advisory-lock guard prevents N× cadence on
// multi-replica deployments.  ** SUPERSEDED BY v4 Section 5a (sha256-derived
// key, extracted to its own module). Kept for diff narrative. **
const LEASES_SWEEP_ADVISORY_KEY = 0x1303_5_5_5_5_5_5_5_5n; // SUPERSEDED

function scheduleSweep(): void {
  setTimeout(async () => {
    let gotLock = false;
    const pool = getDbPool();
    const client = await pool.connect().catch(() => null);
    try {
      if (!client) throw new Error('db pool exhausted at sweep time');
      // Try to acquire advisory lock — only one replica wins per cycle.
      // Released on disconnect (we release explicitly to be tidy).
      const r = await client.query<{ acquired: boolean }>(
        `SELECT pg_try_advisory_lock($1::bigint) AS acquired`,
        [LEASES_SWEEP_ADVISORY_KEY.toString()],
      );
      gotLock = r.rows[0]?.acquired === true;
      if (!gotLock) {
        logger.info({ event: 'leases_sweep_skip_not_leader' }, 'another replica scheduled sweep this cycle');
        return;
      }
      const result = await enqueueJob({
        job_type: 'leases.sweep',
        payload: { grace_minutes: 60 },
        correlation_id: `sweep-${Date.now()}`,
      });
      logger.info({ event: 'leases_sweep_enqueued', job_id: result.job_id, backend: result.backend }, 'leases.sweep scheduled');
    } catch (err) {
      logger.error({ event: 'leases_sweep_enqueue_failed', err: err instanceof Error ? err.message : String(err) }, 'leases.sweep enqueue failed; will retry next interval');
    } finally {
      if (gotLock && client) {
        await client.query(`SELECT pg_advisory_unlock($1::bigint)`, [LEASES_SWEEP_ADVISORY_KEY.toString()]).catch(() => {});
      }
      if (client) client.release();
      scheduleSweep();  // chain next iteration regardless of success
    }
  }, SWEEP_INTERVAL_MS);
}

// In main() after applyMigrations() / bootstrapKgIfEnabled():
scheduleSweep();
logger.info({ interval_minutes: SWEEP_INTERVAL_MS / 60_000 }, 'leases.sweep scheduler started');
```

**v3 advisory-lock semantics:**
- `pg_try_advisory_lock(bigint)` is session-level. We use one connection per attempt, release explicitly on success, drop the connection on failure (released by Postgres on disconnect).
- If a stuck process holds the lock indefinitely (process crash without release), Postgres releases it when the underlying TCP connection drops — typically within seconds. Worst case: one missed cycle every ~5 minutes during crash, recovered automatically.
- The lock is NOT held during the sweep itself (the actual DELETE runs in the worker). The lock only guards "who enqueues the job for this cycle".
- In single-replica deployments this adds ~1ms per sweep attempt — negligible.

Notes:
- Chained setTimeout (NOT setInterval) — guarantees no overlap if a sweep ever takes >15 min.
- First fire after 15 min (not immediate). Rationale v2 (post-Adversary r1): the sweep is **not redundant** with the lazy step-1 DELETE inside `claimArtifact` — lazy cleanup only fires when an agent attempts to re-claim the same artifact_id. The sweep is the only path that removes expired rows for abandoned artifacts (no re-claim). Skipping the immediate-at-startup fire avoids piling work onto cold boot; the first cleanup at +15 min is plenty for steady-state hygiene.
- Errors are logged and swallowed — never crash the server because of a sweep failure.
- Timer holds no exported state. SIGINT/exit clears it; hot-reload isn't supported in this codebase today.

### 10. NEW: `GET /api/me` endpoint (v4 — role + scope + key_source)

`src/api/routes/me.ts` (new file):

```typescript
import { Router } from 'express';
import { getEnv } from '../../core/index.js';

/**
 * GET /api/me — return identity context for the current authenticated user.
 *
 * Returns:
 *   role: 'reader' | 'writer' | 'admin'           (always set)
 *   project_scope: string | null                   (project_id restriction, null = global)
 *   auth_enabled: boolean                           (MCP_AUTH_ENABLED state)
 *   key_source: 'no_auth' | 'env_token' | 'db_key' (v4 — disambiguates 3 admin identities)
 *
 * Resolution (mirrors bearerAuth middleware logic exactly):
 *   - MCP_AUTH_ENABLED=false → role='admin', project_scope=null, key_source='no_auth'
 *   - MCP_AUTH_ENABLED=true + env-var token → role='admin', project_scope=null, key_source='env_token'
 *     (middleware skips role assignment, but env_token is implicitly admin)
 *   - MCP_AUTH_ENABLED=true + DB-backed key → role=apiKeyRole, project_scope=apiKeyScope, key_source='db_key'
 *
 * v4 rationale (post-Adversary r3 BLOCK 1): the three admin identities used to
 * collapse to the same wire shape, making "auth_enabled=false in prod" invisible
 * in the GUI. The key_source field lets the client display warnings when in
 * unprotected modes.
 */
const router = Router();

router.get('/', (req, res) => {
  const env = getEnv();
  const attachedRole = (req as { apiKeyRole?: string }).apiKeyRole;
  const attachedScope = (req as { apiKeyScope?: string | null }).apiKeyScope;
  const authEnabled = env.MCP_AUTH_ENABLED ?? false;

  let role: 'reader' | 'writer' | 'admin';
  let project_scope: string | null;
  let key_source: 'no_auth' | 'env_token' | 'db_key';

  if (!authEnabled) {
    // Anonymous in dev mode — every action is admin-equivalent.
    role = 'admin';
    project_scope = null;
    key_source = 'no_auth';
  } else if (attachedRole === undefined) {
    // bearerAuth returned next() without setting role → env-var token path
    // (the only auth-enabled path that doesn't set apiKeyRole).
    role = 'admin';
    project_scope = null;
    key_source = 'env_token';
  } else {
    role = attachedRole as 'reader' | 'writer' | 'admin';
    project_scope = attachedScope ?? null;
    key_source = 'db_key';
  }

  res.json({ role, project_scope, auth_enabled: authEnabled, key_source });
});

export const meRouter = router;
```

Mount in `src/api/index.ts` after the existing routers:
```typescript

// ...

```

No new DB queries; no migration; no security risk (endpoint reveals only the caller's own role+scope, derived from the bearer the caller supplied).

The GUI does the right thing by scope-filtering; backend enforcement is tracked as **DEFERRED-004** (trigger: Sprint 13.7 E2E suite includes a cross-tenant admin attempt; or any prior security audit identifies the same).
